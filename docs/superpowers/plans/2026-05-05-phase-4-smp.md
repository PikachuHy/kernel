# Phase 4: SMP Bringup Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Bring all CPU cores online via ACPI MADT discovery and manual INIT-SIPI with a custom trampoline.

**Architecture:** Parse the ACPI MADT table (reached via Limine's RSDP → XSDT) to enumerate local APIC entries. Build a trampoline (real-mode → long-mode) in a page below 1MB. Send INIT-SIPI via the LAPIC ICR register to each AP. APs run the trampoline, enter long mode, and jump to `smp_ap_entry()` which sets up per-CPU GDT/stack/LAPIC then parks.

**Tech Stack:** ACPI MADT parsing, x86-64 trampoline (real-mode + protected-mode + long-mode) in GAS assembly, LAPIC ICR for IPI delivery

```cpp
#pragma once

#include <stdint.h>

// ACPI 2.0+ structures for MADT discovery

struct [[gnu::packed]] RSDP {
    char     signature[8];   // "RSD PTR "
    uint8_t  checksum;
    char     oem_id[6];
    uint8_t  revision;
    uint32_t rsdt_addr;      // 32-bit RSDT pointer (ACPI 1.0)
    // ACPI 2.0+ fields:
    uint32_t length;
    uint64_t xsdt_addr;      // 64-bit XSDT pointer
    uint8_t  extended_checksum;
    uint8_t  reserved[3];
};

struct [[gnu::packed]] SDTHeader {
    char     signature[4];
    uint32_t length;
    uint8_t  revision;
    uint8_t  checksum;
    char     oem_id[6];
    char     oem_table_id[8];
    uint32_t oem_revision;
    uint32_t creator_id;
    uint32_t creator_revision;
};

struct [[gnu::packed]] MADT {
    SDTHeader header;
    uint32_t  lapic_addr;
    uint32_t  flags;          // 1 = dual 8259 PICs present
    // Followed by variable-length entries:
    // uint8_t type;
    // uint8_t length;
    // ... type-specific data ...
};

// MADT entry types
constexpr uint8_t MADT_LAPIC  = 0x00;  // Local APIC
constexpr uint8_t MADT_IOAPIC = 0x01;  // I/O APIC
constexpr uint8_t MADT_ISO    = 0x02;  // Interrupt Source Override
constexpr uint8_t MADT_NMI    = 0x04;  // NMI source
constexpr uint8_t MADT_LAPIC_OVERRIDE = 0x05;  // LAPIC address override
constexpr uint8_t MADT_X2APIC = 0x09;  // x2APIC entry

struct [[gnu::packed]] MADTEntry {
    uint8_t type;
    uint8_t length;
};

struct [[gnu::packed]] MADTLapic {
    uint8_t  type;      // 0x00
    uint8_t  length;    // 8
    uint8_t  acpi_cpu_id;
    uint8_t  apic_id;
    uint32_t flags;      // bit 0 = enabled
};

// Maximum CPUs from MADT (QEMU -smp supports up to 255 logical)
constexpr int ACPI_MAX_CPUS = 64;

// Result of MADT parse
struct CpuInfo {
    uint8_t acpi_cpu_id;
    uint8_t lapic_id;
    bool    enabled;
};

// Parse RSDP → XSDT → MADT, fill cpu_info[].
// Returns number of CPUs found (0 on failure).
// hhdm is needed to access ACPI tables (they're at physical addresses).
int acpi_parse_madt(uint64_t hhdm, CpuInfo* cpus, int max_cpus);
```

```cpp
#include "kernel/arch/x86_64/acpi.hpp"
#include "kernel/lib/klog.hpp"

namespace {

inline void* p2v(uint64_t hhdm, uint64_t phys) {
    return reinterpret_cast<void*>(hhdm + phys);
}

bool checksum_ok(const uint8_t* data, size_t len) {
    uint8_t sum = 0;
    for (size_t i = 0; i < len; i++) sum += data[i];
    return sum == 0;
}

} // namespace

int acpi_find_madt(uint64_t hhdm, uint64_t rsdp_phys, const MADT** out_madt) {
    if (!rsdp_phys) return -1;

    auto* rsdp = static_cast<const RSDP*>(p2v(hhdm, rsdp_phys));

    // Validate RSDP signature
    if (rsdp->signature[0] != 'R' || rsdp->signature[1] != 'S' ||
        rsdp->signature[2] != 'D' || rsdp->signature[3] != ' ' ||
        rsdp->signature[4] != 'P' || rsdp->signature[5] != 'T' ||
        rsdp->signature[6] != 'R' || rsdp->signature[7] != ' ') {
        return -1;
    }

    // Validate RSDP checksum
    if (!checksum_ok(reinterpret_cast<const uint8_t*>(rsdp), 20)) {
        return -1;
    }

    // Prefer XSDT (64-bit) on ACPI 2.0+ systems
    const void* root = nullptr;
    uint32_t entry_count = 0;

    if (rsdp->revision >= 2 && rsdp->xsdt_addr) {
        auto* xsdt = static_cast<const SDTHeader*>(p2v(hhdm, rsdp->xsdt_addr));
        if (checksum_ok(reinterpret_cast<const uint8_t*>(xsdt), xsdt->length)) {
            root = xsdt + 1;  // entries follow header
            entry_count = (xsdt->length - sizeof(SDTHeader)) / sizeof(uint64_t);
        }
    }

    if (!root && rsdp->rsdt_addr) {
        auto* rsdt = static_cast<const SDTHeader*>(p2v(hhdm, rsdp->rsdt_addr));
        if (checksum_ok(reinterpret_cast<const uint8_t*>(rsdt), rsdt->length)) {
            root = rsdt + 1;
            entry_count = (rsdt->length - sizeof(SDTHeader)) / sizeof(uint32_t);
        }
    }

    if (!root || !entry_count) return -2;

    // Search for MADT ("APIC")
    for (uint32_t i = 0; i < entry_count; i++) {
        const SDTHeader* hdr;
        if (rsdp->revision >= 2 && rsdp->xsdt_addr) {
            uint64_t addr = static_cast<const uint64_t*>(root)[i];
            hdr = static_cast<const SDTHeader*>(p2v(hhdm, addr));
        } else {
            uint32_t addr = static_cast<const uint32_t*>(root)[i];
            hdr = static_cast<const SDTHeader*>(p2v(hhdm, addr));
        }

        if (hdr->signature[0] == 'A' && hdr->signature[1] == 'P' &&
            hdr->signature[2] == 'I' && hdr->signature[3] == 'C') {
            if (checksum_ok(reinterpret_cast<const uint8_t*>(hdr), hdr->length)) {
                *out_madt = reinterpret_cast<const MADT*>(hdr);
                return 0;
            }
        }
    }

    return -3;  // MADT not found
}

int acpi_parse_cpus(const MADT* madt, CpuInfo* cpus, int max_cpus) {
    if (!madt || !cpus) return 0;

    const uint8_t* start = reinterpret_cast<const uint8_t*>(madt + 1);
    const uint8_t* end   = reinterpret_cast<const uint8_t*>(madt) + madt->header.length;
    int count = 0;

    for (const uint8_t* p = start; p < end; ) {
        const auto* entry = reinterpret_cast<const MADTEntry*>(p);
        if (entry->length == 0) break;  // safety

        if (entry->type == MADT_LAPIC && entry->length >= 8) {
            const auto* lapic = reinterpret_cast<const MADTLapic*>(p);
            if (count < max_cpus) {
                cpus[count].acpi_cpu_id = lapic->acpi_cpu_id;
                cpus[count].lapic_id    = lapic->apic_id;
                cpus[count].enabled     = (lapic->flags & 1) != 0;
                count++;
            }
        }

        p += entry->length;
    }

    return count;
}
```

### Task 3: SMP Trampoline

**Files:**
- Create: `kernel/arch/x86_64/trampoline.S`

The trampoline is loaded at a physical page below 1MB. APs start in 16-bit real mode, transition through 32-bit protected mode, then enter 64-bit long mode, and finally call `smp_ap_entry(cpu_id, lapic_id)`.

Data passed via a small struct at the start of the trampoline page:

```asm
# kernel/arch/x86_64/trampoline.S

.section .trampoline, "ax"
.code16

# ── Trampoline data (first few bytes, read by 16-bit code) ──
.globl trampoline_start
trampoline_start:
    ljmp $0, $real_start    # 5 bytes — jump to real_start

.align 8
trampoline_cr3:   .quad 0   # CR3 value (shared kernel page tables)
trampoline_entry: .quad 0   # 64-bit entry point = smp_ap_entry
trampoline_id:    .quad 0   # arg: cpu_id (low 32) | lapic_id (high 32)
trampoline_gdtr:  .short 0  # GDTR limit (filled at runtime)
trampoline_gdt:   .quad 0   # GDTR base (filled at runtime)

real_start:
    cli
    xorw %ax, %ax
    movw %ax, %ds

    # Load GDT for protected mode
    lgdtl trampoline_gdtr - trampoline_start

    # Enable protected mode (set PE bit in CR0)
    movl %cr0, %eax
    orl  $1, %eax
    movl %eax, %cr0

    # Far jump into 32-bit protected mode
    ljmp $0x08, $protected_start

.code32
protected_start:
    movw $0x10, %ax
    movw %ax, %ds
    movw %ax, %es
    movw %ax, %ss

    # Enable PAE (set PAE bit in CR4)
    movl %cr4, %eax
    orl  $(1 << 5), %eax
    movl %eax, %cr4

    # Load page tables (CR3)
    movl trampoline_cr3 - trampoline_start, %eax
    movl %eax, %cr3

    # Enable long mode (EFER.LME)
    movl $0xC0000080, %ecx
    rdmsr
    orl  $(1 << 8), %eax
    wrmsr

    # Enable paging (set PG bit in CR0)
    movl %cr0, %eax
    orl  $(1 << 31), %eax
    movl %eax, %cr0

    # Far jump into 64-bit long mode
    ljmp $0x18, $long_start

.code64
long_start:
    movw $0x20, %ax
    movw %ax, %ds
    movw %ax, %es
    movw %ax, %ss

    # Load entry point and call smp_ap_entry
    movq trampoline_entry(%rip), %rax
    movq trampoline_id(%rip), %rdi   # rdi = packed cpu_id | lapic_id
    callq *%rax

    # Should never return
1:  hlt
    jmp 1b

.globl trampoline_end
trampoline_end:
```

### Task 4: smp_init() — BSP Bringup Logic

**smp_init flow:**
1. Call `acpi_find_madt()` to get the MADT table
2. Call `acpi_parse_cpus()` to build CPU list
3. Allocate trampoline page below 1MB via bitmap_alloc_page (early allocator gives low pages first)
4. Copy trampoline code to the trampoline page
5. Fill trampoline data: CR3, entry point, GDT
6. For each AP (skip BSP, which is CPU 0):
   a. Set trampoline_id to (cpu_id | lapic_id << 32)
   b. Send INIT IPI via LAPIC ICR
   c. Wait 10ms (PIT or busy-wait)
   d. Send SIPI (Start-up IPI) with trampoline page number as vector
   e. Wait for AP to signal online
7. Return count of online CPUs

```cpp
constexpr uint16_t LAPIC_ICR_LO = 0x300;
constexpr uint16_t LAPIC_ICR_HI = 0x310;

constexpr uint32_t ICR_INIT       = 0x00004500;
constexpr uint32_t ICR_STARTUP    = 0x00004600;
constexpr uint32_t ICR_ASSERT     = 0x00004000;
constexpr uint32_t ICR_DEST_SELF  = 0x00040000;

void lapic_send_ipi(uint32_t lapic_id, uint32_t icr_lo);
void lapic_send_init(uint32_t lapic_id);
void lapic_send_sipi(uint32_t lapic_id, uint8_t vector);  // vector = page number
```

Add to `kernel/arch/x86_64/apic.cpp`:
```cpp
void lapic_send_ipi(uint32_t lapic_id, uint32_t icr_lo) {
    // Set destination
    lapic_write(LAPIC_ICR_HI, lapic_id << 24);
    // Trigger send
    lapic_write(LAPIC_ICR_LO, icr_lo);
    // Wait for delivery
    while (lapic_read(LAPIC_ICR_LO) & (1 << 12)) {
        asm volatile("pause" ::: "memory");
    }
}

void lapic_send_init(uint32_t lapic_id) {
    lapic_send_ipi(lapic_id, ICR_INIT | ICR_ASSERT);
    // 10ms delay for INIT deassert
    for (volatile int i = 0; i < 1000000; i++) asm volatile("pause" ::: "memory");
    lapic_send_ipi(lapic_id, ICR_INIT);  // deassert
}

void lapic_send_sipi(uint32_t lapic_id, uint8_t vector) {
    lapic_send_ipi(lapic_id, ICR_STARTUP | vector);
}
```
