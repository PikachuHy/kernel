> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** IPC-native virtual filesystem with ring-3 FS servers (devfs + tmpfs), kernel mount namespace, and file-as-Channel model.

**Architecture:** Kernel provides mount namespace + sys_open/sys_mount. FS servers are ring-3 processes. File I/O is direct Channel IPC (channel_read/channel_write). Files are Channels — no new KernelObject type.

**Tech Stack:** C++26 freestanding, x86-64, Bazel 9, llvm-objcopy for ELF embedding

```cpp
#pragma once

// File operation message sent over a file's Channel.
// For Open (on the mount Channel): path follows in the data payload.
struct FileMsg {
    enum Op : uint32_t {
        Open    = 0,
        Read    = 1,
        Write   = 2,
        Seek    = 3,
        Stat    = 4,
        Close   = 5,
        Readdir = 6,
    };
    Op       op;
    uint32_t flags;      // open flags (O_CREAT, O_RDONLY, O_WRONLY, O_RDWR)
    uint64_t offset;     // seek offset, or readdir cookie
    uint64_t length;     // bytes to read/write
};

// Open flags
constexpr uint32_t O_RDONLY = 1 << 0;
constexpr uint32_t O_WRONLY = 1 << 1;
constexpr uint32_t O_RDWR   = 1 << 2;
constexpr uint32_t O_CREAT  = 1 << 3;

// Response from FS server for any file operation.
struct FileResponse {
    int32_t  result;     // 0 = success, negative = errno
    uint64_t size;       // bytes transferred, file size for stat, new offset for seek
};

struct Stat {
    uint64_t size;
    uint32_t type;       // 0 = file, 1 = directory
    uint32_t padding;
};

struct Dirent {
    char     name[256];
    uint32_t type;       // 0 = file, 1 = directory
    uint64_t size;
};
```

```cpp
#pragma once
#include <stdint.h>
#include <stddef.h>

class Channel;
class Process;

struct MountEntry {
    char     path[256];
    Channel* fs_channel;   // kernel-end of the mount Channel (raw pointer)
    Process* fs_process;   // the FS server process (for handle allocation)
    bool     active;
};

constexpr size_t MAX_MOUNTS = 16;

// Initialize the mount table.
void mount_init();

// Add or replace a mount entry. The kernel takes ownership of fs_channel.
// Returns 0 on success, -1 if table is full.
int mount_add(const char* path, Channel* fs_channel, Process* fs_process);

// Remove a mount entry by path. Returns 0 on success, -1 if not found.
int mount_remove(const char* path);

// Find the longest-prefix matching mount entry for the given path.
// Returns nullptr if no match.
MountEntry* mount_resolve(const char* path);
```

- [ ] **Step 6: Implement mount namespace**

Create `kernel/fs/mount.cpp`:

```cpp
#include "kernel/fs/mount.hpp"
#include "kernel/core/object/channel.hpp"
#include "kernel/core/object/process.hpp"

static MountEntry g_mounts[MAX_MOUNTS];
static size_t     g_mount_count = 0;

void mount_init() {
    for (size_t i = 0; i < MAX_MOUNTS; i++) {
        g_mounts[i].active = false;
        g_mounts[i].fs_channel = nullptr;
        g_mounts[i].fs_process = nullptr;
        g_mounts[i].path[0] = '\0';
    }
    g_mount_count = 0;
}

int mount_add(const char* path, Channel* fs_channel, Process* fs_process) {
    if (!path || !fs_channel || !fs_process) return -1;

    // Replace existing entry if present
    for (size_t i = 0; i < MAX_MOUNTS; i++) {
        if (g_mounts[i].active) {
            const char* a = g_mounts[i].path;
            const char* b = path;
            bool same = true;
            while (*a && *b) { if (*a != *b) { same = false; break; } a++; b++; }
            if (same && *a == '\0' && *b == '\0') {
                g_mounts[i].fs_channel = fs_channel;
                g_mounts[i].fs_process = fs_process;
                return 0;
            }
        }
    }

    // Add new entry
    if (g_mount_count >= MAX_MOUNTS) return -1;
    size_t i = g_mount_count;
    size_t j = 0;
    while (path[j] && j < 255) { g_mounts[i].path[j] = path[j]; j++; }
    g_mounts[i].path[j] = '\0';
    g_mounts[i].fs_channel = fs_channel;
    g_mounts[i].fs_process = fs_process;
    g_mounts[i].active = true;
    g_mount_count++;
    return 0;
}

int mount_remove(const char* path) {
    for (size_t i = 0; i < MAX_MOUNTS; i++) {
        if (!g_mounts[i].active) continue;
        const char* a = g_mounts[i].path;
        const char* b = path;
        bool same = true;
        while (*a && *b) { if (*a != *b) { same = false; break; } a++; b++; }
        if (same && *a == '\0' && *b == '\0') {
            g_mounts[i].active = false;
            g_mounts[i].fs_channel = nullptr;
            g_mounts[i].fs_process = nullptr;
            g_mount_count--;
            return 0;
        }
    }
    return -1;
}

MountEntry* mount_resolve(const char* path) {
    MountEntry* best = nullptr;
    size_t best_len = 0;
    for (size_t i = 0; i < MAX_MOUNTS; i++) {
        if (!g_mounts[i].active) continue;
        const char* prefix = g_mounts[i].path;
        // Check if prefix is a path-prefix of path
        size_t j = 0;
        while (prefix[j] && path[j] && prefix[j] == path[j]) j++;
        if (prefix[j] == '\0' && (path[j] == '/' || path[j] == '\0')) {
            if (j > best_len) {
                best = &g_mounts[i];
                best_len = j;
            }
        }
    }
    return best;
}
```

```cpp
uint64_t sys_channel_read(uint64_t a1, uint64_t a2, uint64_t a3, uint64_t a4) {
    Process* proc = current_process();
    if (!proc) return static_cast<uint64_t>(-1);

    handle_t h = static_cast<handle_t>(a1);
    KernelObject* obj = proc->handles.Lookup(h, Rights{.mask = Rights::Read});
    if (!obj || obj->type() != KernelObject::Type::Channel) return static_cast<uint64_t>(-1);

    size_t out_len = 0;
    size_t out_handles = 0;
    int rc;

    // Block until a message is available
    while (true) {
        rc = static_cast<Channel*>(obj)->Read(
            reinterpret_cast<void*>(a2), a3, &out_len,
            nullptr, 0, &out_handles);
        if (rc != -2) break;  // -2 = empty
        thread_yield();       // give up CPU, wait for message
    }

    return (rc == 0) ? out_len : static_cast<uint64_t>(rc);
}

uint64_t sys_channel_read_handles(uint64_t a1, uint64_t a2, uint64_t a3, uint64_t a4) {
    Process* proc = current_process();
    if (!proc) return static_cast<uint64_t>(-1);

    handle_t h = static_cast<handle_t>(a1);
    KernelObject* obj = proc->handles.Lookup(h, Rights{.mask = Rights::Read});
    if (!obj || obj->type() != KernelObject::Type::Channel) return static_cast<uint64_t>(-1);

    // Args packed in struct (4-register ABI can't pass 5 scalars)
    struct ChannelReadHandleArgs {
        handle_t* handle_buf;
        size_t    buf_capacity;
    };
    auto* args = reinterpret_cast<const ChannelReadHandleArgs*>(a2);

    size_t out_len = 0;
    size_t out_handles = 0;
    int rc;

    // Block until a message is available
    while (true) {
        rc = static_cast<Channel*>(obj)->Read(
            reinterpret_cast<void*>(a3), a4, &out_len,
            args ? args->handle_buf : nullptr,
            args ? args->buf_capacity : 0,
            &out_handles);
        if (rc != -2) break;  // -2 = empty
        thread_yield();
    }

    if (rc == 0) return (out_handles << 32) | (out_len & 0xFFFFFFFFULL);
    return static_cast<uint64_t>(rc);
}
```

```cpp
// ── VFS syscalls ─────────────────────────────────────────────────

uint64_t sys_open(uint64_t a1, uint64_t a2, uint64_t, uint64_t) {
    const char* path = reinterpret_cast<const char*>(a1);
    uint64_t flags = a2;

    MountEntry* mount = mount_resolve(path);
    if (!mount) return INVALID_HANDLE;

    // Compute the relative path after the mount prefix
    const char* prefix = mount->path;
    while (*prefix) { path++; prefix++; }
    if (*path == '/') path++;

    // Trim trailing slash if present and not root
    size_t path_len = 0;
    while (path[path_len]) path_len++;

    // Create the file Channel pair
    Channel* file_chan = static_cast<Channel*>(kmalloc(sizeof(Channel)));
    if (!file_chan) return INVALID_HANDLE;
    new (file_chan) Channel();
    // refcount starts at 1

    // Allocate a handle for the file Channel in the FS server's process
    // so the FS server can reference it
    Process* fs_proc = mount->fs_process;
    Rights full_rights{.mask = Rights::Read | Rights::Write |
                               Rights::Duplicate | Rights::Transfer};
    handle_t fs_file_handle = fs_proc->handles.Alloc(file_chan, full_rights);
    // fs_proc->handles.Alloc calls AddRef; we still have the initial ref

    // Build the Open message
    FileMsg msg{};
    msg.op = FileMsg::Open;
    msg.flags = static_cast<uint32_t>(flags);
    msg.offset = 0;
    msg.length = 0;

    // Send Open request to FS server (with path as data payload).
    // The FS server gets the fs_file_handle value in the message and uses it.
    // We encode the handle_t into the data payload for the FS server to receive.
    struct OpenPayload {
        char     path[256];
        handle_t file_handle;
    };
    OpenPayload payload{};
    size_t i = 0;
    while (path[i] && i < 255) { payload.path[i] = path[i]; i++; }
    payload.path[i] = '\0';
    payload.file_handle = fs_file_handle;

    int rc = mount->fs_channel->Write(&payload, sizeof(payload), nullptr, 0);
    if (rc != 0) {
        fs_proc->handles.Free(fs_file_handle);
        file_chan->Release();
        return INVALID_HANDLE;
    }

    // Wait for the FS server's response on the mount Channel
    FileResponse resp;
    size_t out_len;
    while (true) {
        rc = mount->fs_channel->Read(&resp, sizeof(resp), &out_len,
                                      nullptr, 0, nullptr);
        if (rc != -2) break;
        thread_yield();
    }

    if (rc != 0 || resp.result != 0) {
        fs_proc->handles.Free(fs_file_handle);
        file_chan->Release();
        return INVALID_HANDLE;
    }

    // Allocate a handle for the caller's process
    Process* cur = current_process();
    if (!cur) {
        fs_proc->handles.Free(fs_file_handle);
        file_chan->Release();
        return INVALID_HANDLE;
    }

    handle_t client_handle = cur->handles.Alloc(file_chan, full_rights);

    // The FS server's handle + client's handle each hold a ref.
    // Release our temporary ref (from construction). The two handles
    // keep the object alive.
    file_chan->Release();

    return client_handle;
}

uint64_t sys_mount(uint64_t a1, uint64_t a2, uint64_t, uint64_t) {
    const char* path = reinterpret_cast<const char*>(a1);
    handle_t h = static_cast<handle_t>(a2);

    Process* cur = current_process();
    if (!cur) return static_cast<uint64_t>(-1);

    KernelObject* obj = cur->handles.Lookup(h);
    if (!obj || obj->type() != KernelObject::Type::Channel) return static_cast<uint64_t>(-1);

    Channel* ch = static_cast<Channel*>(obj);

    // Determine the FS server process from the current thread
    Process* fs_proc = current_thread()->process;

    return static_cast<uint64_t>(mount_add(path, ch, fs_proc));
}
```

```cpp
// kernel/fs/devfs/devfs.cpp
// devfs filesystem server — ring-3 process
//
// Serves: /dev/null, /dev/zero, /dev/console
// Receives FileMsg via handle 0 (mount Channel).
// Open creates a new Channel pair for the file.

using uint64_t = unsigned long long;
using uint32_t = unsigned int;
using int32_t  = int;
using uint8_t  = unsigned char;
using size_t = decltype(sizeof(0));

// ── Syscall numbers (must match kernel) ──────────────────────────
constexpr int SYS_DEBUG_PRINT          = 0;
constexpr int SYS_CHANNEL_WRITE        = 11;
constexpr int SYS_CHANNEL_READ         = 12;
constexpr int SYS_HANDLE_CLOSE         = 1;
constexpr int SYS_PROCESS_EXIT         = 31;

// ── Protocol types (must match kernel/fs/protocol.hpp) ──────────
struct FileMsg {
    enum Op : uint32_t { Open=0, Read=1, Write=2, Seek=3, Stat=4, Close=5, Readdir=6 };
    Op       op;
    uint32_t flags;
    uint64_t offset;
    uint64_t length;
};

struct FileResponse {
    int32_t  result;
    uint64_t size;
};

struct Stat {
    uint64_t size;
    uint32_t type;
    uint32_t padding;
};

    uint32_t file_handle;

// ── Syscall wrappers ─────────────────────────────────────────────
static uint64_t syscall6(uint64_t num, uint64_t a1, uint64_t a2,
                          uint64_t a3, uint64_t a4, uint64_t a5) {
    uint64_t ret;
    asm volatile(
        "movq %1, %%rax\n"
        "movq %2, %%rdi\n"
        "movq %3, %%rsi\n"
        "movq %4, %%rdx\n"
        "movq %5, %%r10\n"
        "movq %6, %%r8\n"
        "syscall\n"
        "movq %%rax, %0\n"
        : "=r"(ret)
        : "r"(num), "r"(a1), "r"(a2), "r"(a3), "r"(a4), "r"(a5)
        : "rax", "rdi", "rsi", "rdx", "r10", "r8", "rcx", "r11", "memory"
    );
    return ret;
}

static void debug(const char* msg) {
    syscall6(SYS_DEBUG_PRINT, (uint64_t)msg, 0, 0, 0, 0);
}

static int channel_write(uint32_t h, const void* data, size_t len) {
    struct WriteArgs { const void* data; size_t len; const uint32_t* handles; size_t num; };
    WriteArgs args = { data, len, nullptr, 0 };
    return (int)syscall6(SYS_CHANNEL_WRITE, h, (uint64_t)&args, 0, 0, 0);
}

static int channel_read(uint32_t h, void* buf, size_t buf_size) {
    return (int)syscall6(SYS_CHANNEL_READ, h, (uint64_t)buf, buf_size, 0, 0);
}

// ── Device implementations ────────────────────────────────────────
// Each handler reads FileMsg (optionally + data) from file_chan,
// processes the request, and writes FileResponse (+ data if read/stat) back.

static void handle_null(uint32_t file_chan) {
    while (true) {
        uint8_t buf[4096];
        int rc = channel_read(file_chan, buf, sizeof(buf));
        if (rc < 0) break;

        if ((size_t)rc < sizeof(FileMsg)) continue;
        FileMsg* msg = reinterpret_cast<FileMsg*>(buf);

        FileResponse resp = {};
        switch (msg->op) {
        case FileMsg::Read:
            resp.result = 0;
            resp.size = 0;  // EOF
            channel_write(file_chan, &resp, sizeof(resp));
            break;
        case FileMsg::Write:
            resp.result = 0;
            resp.size = msg->length;
            channel_write(file_chan, &resp, sizeof(resp));
            break;
        case FileMsg::Stat: {
            Stat st = {0, 0, 0};
            uint8_t out[sizeof(FileResponse) + sizeof(Stat)];
            *(FileResponse*)out = {0, sizeof(st)};
            *(Stat*)(out + sizeof(FileResponse)) = st;
            channel_write(file_chan, out, sizeof(out));
            break;
        }
        case FileMsg::Close:
            handle_close(file_chan);
            return;
        default:
            resp.result = -1;
            channel_write(file_chan, &resp, sizeof(resp));
            break;
        }
    }
}

static void handle_zero(uint32_t file_chan) {
    while (true) {
        uint8_t buf[4096];
        int rc = channel_read(file_chan, buf, sizeof(buf));
        if (rc < 0) break;

        if ((size_t)rc < sizeof(FileMsg)) continue;
        FileMsg* msg = reinterpret_cast<FileMsg*>(buf);

        FileResponse resp = {};
        switch (msg->op) {
        case FileMsg::Read: {
            size_t count = msg->length < (4096 - sizeof(FileResponse))
                               ? msg->length : (4096 - sizeof(FileResponse));
            uint8_t out[sizeof(FileResponse) + 4096];
            *(FileResponse*)out = {0, count};
            for (size_t i = 0; i < count; i++)
                out[sizeof(FileResponse) + i] = 0;
            channel_write(file_chan, out, sizeof(FileResponse) + count);
            break;
        }
        case FileMsg::Write:
            resp.result = 0;
            resp.size = msg->length;
            channel_write(file_chan, &resp, sizeof(resp));
            break;
        case FileMsg::Stat: {
            Stat st = {0, 0, 0};
            uint8_t out[sizeof(FileResponse) + sizeof(Stat)];
            *(FileResponse*)out = {0, sizeof(st)};
            *(Stat*)(out + sizeof(FileResponse)) = st;
            channel_write(file_chan, out, sizeof(out));
            break;

        case FileMsg::Close:
            handle_close(file_chan);
            return;
        default:
            resp.result = -1;
            channel_write(file_chan, &resp, sizeof(resp));
            break;
        }
    }
}

static void handle_console(uint32_t file_chan) {
    while (true) {
        uint8_t buf[4096];
        int rc = channel_read(file_chan, buf, sizeof(buf));
        if (rc < 0) break;

        if ((size_t)rc < sizeof(FileMsg)) continue;
        FileMsg* msg = reinterpret_cast<FileMsg*>(buf);
        uint8_t* data = buf + sizeof(FileMsg);
        size_t data_len = (size_t)rc - sizeof(FileMsg);

        FileResponse resp = {};
        switch (msg->op) {
        case FileMsg::Write:
            // Write the data to debug output
            for (size_t i = 0; i < data_len && i < msg->length; i++) {

            }

            resp.size = msg->length;
            channel_write(file_chan, &resp, sizeof(resp));

        case FileMsg::Read:

            *(FileResponse*)out = {0, sizeof(st)};
            *(Stat*)(out + sizeof(FileResponse)) = st;

}

// ── Entry point ──────────────────────────────────────────────────
extern "C" void _start() {
    debug("devfs: starting\n");

    // Handle 0 is the server mount Channel (pre-allocated by kernel)
    const uint32_t MOUNT_CHAN = 0;

    while (true) {
        // Read Open request from mount Channel
        // Kernel sends: OpenPayload{path, file_handle}
        // file_handle is pre-allocated in OUR handle table — use it directly.
        OpenPayload payload;
        int rc = channel_read(MOUNT_CHAN, &payload, sizeof(payload));
        if (rc < 0) break;

        debug("devfs: open '");
        debug(payload.path);
        debug("'\n");

        uint32_t server_handle = payload.file_handle;

        // Respond to kernel with success
        FileResponse resp = {0, 0};
        channel_write(MOUNT_CHAN, &resp, sizeof(resp));

        // Handle the file in a loop on the server side
        // Simple dispatch based on path
        const char* p = payload.path;

        // Check path equality manually
        bool is_null = false, is_zero = false, is_console = false;
        {
            const char* s = "null";
            bool match = true;
            for (int i = 0; i < 4; i++) { if (p[i] != s[i]) match = false; }
            if (match && p[4] == '\0') is_null = true;
        }
        {
            const char* s = "zero";
            bool match = true;
            for (int i = 0; i < 4; i++) { if (p[i] != s[i]) match = false; }
            if (match && p[4] == '\0') is_zero = true;
        }
        {
            const char* s = "console";
            bool match = true;
            for (int i = 0; i < 7; i++) { if (p[i] != s[i]) match = false; }
            if (match && p[7] == '\0') is_console = true;
        }

        if (is_null) {
            handle_null(server_handle);
        } else if (is_zero) {
            handle_zero(server_handle);
        } else if (is_console) {
            handle_console(server_handle);
        } else {
            debug("devfs: unknown device\n");
        }

        // Close our end of the file Channel
        syscall6(SYS_HANDLE_CLOSE, server_handle, 0, 0, 0, 0);
    }

    debug("devfs: exiting\n");
    syscall6(SYS_PROCESS_EXIT, 0, 0, 0, 0, 0);
    while (1) { asm volatile("hlt"); }
}
```

```cpp
// kernel/fs/tmpfs/tmpfs.cpp
// tmpfs filesystem server — ring-3 process
//
// In-memory writable filesystem. Files are VMO-backed pages.
// Directories are special VMOs storing Dirent arrays.

using uint64_t = unsigned long long;
using uint32_t = unsigned int;
using int32_t  = int;
using uint8_t  = unsigned char;
using size_t = decltype(sizeof(0));

// ── Syscall numbers ──────────────────────────────────────────────
constexpr int SYS_DEBUG_PRINT          = 0;
constexpr int SYS_CHANNEL_CREATE       = 10;
constexpr int SYS_CHANNEL_WRITE        = 11;
constexpr int SYS_CHANNEL_READ         = 12;
constexpr int SYS_HANDLE_CLOSE         = 1;
constexpr int SYS_PROCESS_EXIT         = 31;
constexpr int SYS_VMO_CREATE           = 40;
constexpr int SYS_VMO_MAP              = 41;

// ── Protocol types ───────────────────────────────────────────────
struct FileMsg {
    enum Op : uint32_t { Open=0, Read=1, Write=2, Seek=3, Stat=4, Close=5, Readdir=6 };
    Op       op;
    uint32_t flags;
    uint64_t offset;
    uint64_t length;
};
struct FileResponse { int32_t result; uint64_t size; };
struct Stat { uint64_t size; uint32_t type; uint32_t padding; };
struct Dirent { char name[256]; uint32_t type; uint64_t size; };
struct OpenPayload { char path[256]; uint32_t file_handle; };

constexpr uint32_t O_RDONLY = 1 << 0;
constexpr uint32_t O_WRONLY = 1 << 1;
constexpr uint32_t O_RDWR   = 1 << 2;
constexpr uint32_t O_CREAT  = 1 << 3;

// ── Syscall wrappers ─────────────────────────────────────────────
static uint64_t syscall6(uint64_t num, uint64_t a1, uint64_t a2,
                          uint64_t a3, uint64_t a4, uint64_t a5) {

        "movq %1, %%rax\n"
        "movq %2, %%rdi\n"
        "movq %3, %%rsi\n"
        "movq %4, %%rdx\n"
        "movq %5, %%r10\n"
        "movq %6, %%r8\n"
        "syscall\n"
        "movq %%rax, %0\n"
        : "=r"(ret)
        : "r"(num), "r"(a1), "r"(a2), "r"(a3), "r"(a4), "r"(a5)
        : "rax", "rdi", "rsi", "rdx", "r10", "r8", "rcx", "r11", "memory"
    );

// ── Simple linked-list directory (in-memory) ─────────────────────
struct DirEntry {
    char     name[256];
    uint32_t vmo_handle;   // handle to the file's VMO (0 if directory)
    bool     is_dir;
    uint64_t size;
    DirEntry* next;
};

static DirEntry* root = nullptr;

static DirEntry* find_entry(const char* name) {
    for (DirEntry* e = root; e; e = e->next) {
        int i = 0;
        while (name[i] && e->name[i] && name[i] == e->name[i]) i++;
        if (name[i] == '\0' && e->name[i] == '\0') return e;
    }
    return nullptr;
}

static DirEntry* add_entry(const char* name, uint32_t vmo_handle, bool is_dir, uint64_t sz) {
    // Allocate from scratch space
    static DirEntry pool[64];
    static int pool_idx = 0;
    if (pool_idx >= 64) return nullptr;
    DirEntry* e = &pool[pool_idx++];
    int i = 0;
    while (name[i] && i < 255) { e->name[i] = name[i]; i++; }
    e->name[i] = '\0';
    e->vmo_handle = vmo_handle;
    e->is_dir = is_dir;
    e->size = sz;
    e->next = root;
    root = e;
    return e;
}

// ── Per-file state ───────────────────────────────────────────────
struct FileState {
    uint32_t  file_chan;   // server-side handle for the file Channel
    uint32_t  vmo_handle;  // VMO backing this file
    uint64_t  cursor;      // current read/write position
    bool      is_dir;
    FileState* next;
};

static FileState* file_list = nullptr;
static FileState pool2[64];
static int pool2_idx = 0;

static FileState* alloc_file_state(uint32_t fchan, uint32_t vmo, bool dir) {
    if (pool2_idx >= 64) return nullptr;
    FileState* fs = &pool2[pool2_idx++];
    fs->file_chan = fchan;
    fs->vmo_handle = vmo;
    fs->cursor = 0;
    fs->is_dir = dir;
    fs->next = file_list;
    file_list = fs;
    return fs;
}

static FileState* find_file_state(uint32_t fchan) {
    for (FileState* fs = file_list; fs; fs = fs->next)
        if (fs->file_chan == fchan) return fs;
    return nullptr;
}

// ── File operation handlers ──────────────────────────────────────

static void handle_file(FileState* fs) {
    // Map the VMO at a fixed address so we can read/write it
    // Use 0x70000000000 as temporary mapping area
    const uint64_t MAP_BASE = 0x70000000000ULL;
    vmo_map(fs->vmo_handle, MAP_BASE, 0x3, 0);  // RW

    while (true) {
        uint8_t buf[4096];
        int rc = channel_read(fs->file_chan, buf, sizeof(buf));
        if (rc < 0) break;

        if ((size_t)rc < sizeof(FileMsg)) continue;
        FileMsg* msg = reinterpret_cast<FileMsg*>(buf);
        uint8_t* data = buf + sizeof(FileMsg);
        size_t data_len = (size_t)rc - sizeof(FileMsg);

        FileResponse resp = {};

        switch (msg->op) {
        case FileMsg::Read: {
            size_t count = msg->length;
            if (count > 4096 - sizeof(FileResponse)) count = 4096 - sizeof(FileResponse);
            if (msg->offset + count > fs->size()) count = fs->size() - msg->offset;

            // Build response + data
            *(FileResponse*)buf = {0, count};
            uint8_t* src = reinterpret_cast<uint8_t*>(MAP_BASE + msg->offset);
            uint8_t* dst = buf + sizeof(FileResponse);
            for (size_t i = 0; i < count; i++) dst[i] = src[i];
            channel_write(fs->file_chan, buf, sizeof(FileResponse) + count);
            break;
        }
        case FileMsg::Write: {
            size_t count = data_len;
            if (msg->offset + count > fs->size()) {
                // TODO: extend VMO
                count = fs->size() > msg->offset ? fs->size() - msg->offset : 0;
            }
            uint8_t* dst = reinterpret_cast<uint8_t*>(MAP_BASE + msg->offset);
            for (size_t i = 0; i < count; i++) dst[i] = data[i];
            resp.result = 0;
            resp.size = count;
            channel_write(fs->file_chan, &resp, sizeof(resp));
            break;
        }
        case FileMsg::Seek:
            fs->cursor = msg->offset;
            resp.result = 0;
            resp.size = fs->cursor;
            channel_write(fs->file_chan, &resp, sizeof(resp));
            break;
        case FileMsg::Stat: {
            Stat st = {fs->size(), fs->is_dir ? 1u : 0u, 0};
            resp.result = 0;
            resp.size = sizeof(st);
            // Send stat as data after response
            uint8_t out[sizeof(FileResponse) + sizeof(Stat)];
            *(FileResponse*)out = resp;
            *(Stat*)(out + sizeof(FileResponse)) = st;
            channel_write(fs->file_chan, out, sizeof(out));
            break;
        }
        case FileMsg::Readdir:
            // For directories, build dirent list
            if (fs->is_dir) {
                resp.result = 0;
                resp.size = 0;  // simplified: no readdir yet
            } else {
                resp.result = -1;
            }
            channel_write(fs->file_chan, &resp, sizeof(resp));
            break;
        case FileMsg::Close:
            handle_close(fs->file_chan);
            return;
        default:
            resp.result = -1;
            channel_write(fs->file_chan, &resp, sizeof(resp));
            break;
        }
    }
}

// ── Entry point ──────────────────────────────────────────────────
extern "C" void _start() {
    debug("tmpfs: starting\n");
    const uint32_t MOUNT_CHAN = 0;

    while (true) {
        OpenPayload payload;
        int rc = channel_read(MOUNT_CHAN, &payload, sizeof(payload));
        if (rc < 0) break;

        debug("tmpfs: open '");
        debug(payload.path);
        debug("'\n");

        // Look up or create the file
        bool is_dir = (payload.path[0] == '\0');  // root = directory
        DirEntry* entry = nullptr;
        if (!is_dir) {
            entry = find_entry(payload.path);
            if (!entry && (payload.flags & O_CREAT)) {
                uint64_t init_size = 4096;
                uint32_t vmo_h = vmo_create(init_size);
                entry = add_entry(payload.path, vmo_h, false, init_size);
                debug("tmpfs: created file\n");
            }
        }

        // Use the pre-allocated file handle from the kernel
        uint32_t server_handle = payload.file_handle;

        // Respond to kernel
        FileResponse resp = {};
        if (is_dir || entry) {
            resp.result = 0;
        } else {
            resp.result = -1;  // ENOENT
        }
        channel_write(MOUNT_CHAN, &resp, sizeof(resp));

        if (resp.result == 0) {
            uint32_t vmo_h = is_dir ? 0 : entry->vmo_handle;
            FileState* fs = alloc_file_state(server_handle, vmo_h, is_dir);
            if (fs) handle_file(fs);
        }

        handle_close(server_handle);
    }

    debug("tmpfs: exiting\n");
    syscall6(SYS_PROCESS_EXIT, 0, 0, 0, 0, 0);
    while (1) { asm volatile("hlt"); }
}
```

Edit `kernel/core/elf_loader.cpp`, add implementation after `elf_load_init_process`:

```cpp
// ── FS server embedding symbols ──────────────────────────────────
extern "C" uint8_t _binary_devfs_bin_start[];
extern "C" uint8_t _binary_devfs_bin_end[];
extern "C" uint8_t _binary_tmpfs_bin_start[];
extern "C" uint8_t _binary_tmpfs_bin_end[];

extern "C" void elf_load_fs_servers() {
    // ── devfs ─────────────────────────────────────────────────
    {
        uint64_t size = _binary_devfs_bin_end - _binary_devfs_bin_start;
        klog("Loading devfs server ("); klog_hex(size); klog(" bytes)...\n");

        Thread* thr = nullptr;
        Process* proc = elf_load(_binary_devfs_bin_start, size, "devfs", 1, &thr);
        if (!proc || !thr) {
            klog("  FAILED to load devfs\n");
            return;
        }

        // Create mount Channel pair for devfs
        Channel* mount_chan = static_cast<Channel*>(kmalloc(sizeof(Channel)));
        if (!mount_chan) {
            klog("  FAILED to create devfs mount Channel\n");
            return;
        }
        new (mount_chan) Channel();

        // Allocate handle 0 in the devfs process for the server end
        Rights full{.mask = Rights::Read | Rights::Write |
                           Rights::Duplicate | Rights::Transfer};
        proc->handles.Alloc(mount_chan, full);
        // mount_chan now has ref for the handle; we still have the initial ref

        // Add to mount table (kernel keeps its ref to mount_chan)
        mount_add("/dev", mount_chan, proc);

        // Start the devfs server thread
        thread_start(thr);
        klog("  devfs server started, mounted at /dev\n");
    }

    // ── tmpfs ─────────────────────────────────────────────────
    {
        uint64_t size = _binary_tmpfs_bin_end - _binary_tmpfs_bin_start;
        klog("Loading tmpfs server ("); klog_hex(size); klog(" bytes)...\n");

        Thread* thr = nullptr;
        Process* proc = elf_load(_binary_tmpfs_bin_start, size, "tmpfs", 1, &thr);
        if (!proc || !thr) {
            klog("  FAILED to load tmpfs\n");
            return;
        }

        Channel* mount_chan = static_cast<Channel*>(kmalloc(sizeof(Channel)));
        if (!mount_chan) {
            klog("  FAILED to create tmpfs mount Channel\n");
            return;
        }
        new (mount_chan) Channel();

        Rights full{.mask = Rights::Read | Rights::Write |
                           Rights::Duplicate | Rights::Transfer};
        proc->handles.Alloc(mount_chan, full);

        mount_add("/", mount_chan, proc);

        thread_start(thr);
        klog("  tmpfs server started, mounted at /\n");
    }
}
```

```cpp
    // Open /dev/console
    uint64_t console_handle = syscall6(SYS_OPEN,

    print("  sys_open(/dev/console): ");

    if ((int64_t)console_handle >= 0) {
        // Write a greeting via VFS
        const char* greeting = "Hello from init via VFS!\n";
        // For channel_write: a1=handle, a2=ptr to ChannelWriteArgs

        WA wa = { greeting, 24, nullptr, 0 };

        print("  channel_write(/dev/console): OK\n");
        syscall6(SYS_HANDLE_CLOSE, console_handle, 0, 0, 0, 0);
```
